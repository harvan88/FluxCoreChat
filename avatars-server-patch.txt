// Patch para agregar ruta /avatars/ en server.ts
// Insertar después de la ruta /uploads/

      // Serve global avatars (Content-Addressable Storage)
      if (url.pathname.startsWith('/avatars/')) {
        const checksum = url.pathname.replace(/^\/avatars\//, '');

        // Validación básica de checksum (16 chars hex)
        if (!/^[a-f0-9]{16}$/.test(checksum)) {
          return new Response('Invalid checksum', { status: 400 });
        }

        // Sharding para distribución
        const shard1 = checksum.slice(0, 2);
        const shard2 = checksum.slice(2, 4);

        const filePath = path.join(
          process.cwd(),
          `uploads/assets/avatars/${shard1}/${shard2}/${checksum}`
        );

        if (!fs.existsSync(filePath)) {
          return new Response('Avatar not found', { status: 404 });
        }

        const file = Bun.file(filePath);

        return new Response(file, {
          headers: {
            'Content-Type': file.type || 'application/octet-stream',
            'Cache-Control': 'public, max-age=31536000, immutable',
          },
        });
      }
